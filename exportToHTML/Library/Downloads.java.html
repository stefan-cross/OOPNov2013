public class Downloads implements IDownloads { 
 
    private int downloads; 
 
    public Downloads(){ 
        downloads = 0; 
    } 
 
    public int getDownloads() { 
        return downloads; 
    } 
 
    public void setDownloads(int i) { 
        downloads = i; 
    } 
 
    public void incDownloads() { 
        downloads++; 
    } 
}
